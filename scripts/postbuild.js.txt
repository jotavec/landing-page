import { copyFileSync } from "fs";
import { mkdirSync } from "fs";
import { existsSync } from "fs";

const from = "../public";
const to = "./dist";

if (!existsSync(to)) mkdirSync(to, { recursive: true });

try {
  copyFileSync(`${from}/sitemap.xml`, `${to}/sitemap.xml`);
  copyFileSync(`${from}/robots.txt`, `${to}/robots.txt`);
  console.log("✅ sitemap.xml e robots.txt copiados para dist/");
} catch (e) {
  console.error("⚠️ Não foi possível copiar sitemap/robots:", e);
}
